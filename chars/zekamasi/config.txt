;_______________________________________________________________________

;凶化型タービン Extra mode2
;_______________________________________________________________________

[State -3, VarSet]
type = VarSet
trigger1 = 1
var(15) = 0  
;<この値を1にするとオルモック耐性、大破逆境補正、対大型目標補正が付与されます。
;◆代償として根性補正が無くなります◆
;オルモック耐性…本体がx方向へのベクトルを持っている場合(やられ時以外)に-2ステートにて無敵付与
;大破逆境補正…大破状態（Life333以下）になると、攻撃力が激的に上昇する
;対大型目標補正…相手のLife及びDefを参照し、規定以上の装甲や体力を持つ場合、攻撃力に補正を掛ける

;要はそれなりに強くなる。ただそれだけ
;近代化改修との併用可

;_______________________________________________________________________

;支援艦隊要請 Extra mode3
;_______________________________________________________________________

[State -3, VarSet]
type = VarSet
trigger1 = 1
var(44) = 0  
;凶化型タービンON時のみ設定可能
;<この値を1にするとLife333以下の時、30F以上継続してやられ状態が続く攻撃を受けると
;ストライカー召喚可能時、天津風が自動砲撃します。

;要はコンボカット機能



;_____________________________________________________________________________
;
;逆根性値設定(プレイヤー操作でも関係あります) attack reduction by enemy's life
;_____________________________________________________________________________
[state -3];逆根性値設定
type = varset
trigger1 = !ishelper
trigger1 = roundstate != 2
var(12) = 1
;0 ---->逆根性値なし off
;1 ---->逆根性値あり on

;注意
;根性値持ちのEFZキャラにはある程度対応して逆根性値をはずしています。
;無敵医師氏のGG勢にも対応しました。
;馬の骨氏の記述をそのままお借りしています。
;逆根性値とは…相手に擬似根性値を設定する項目です。
;要は相手のLifeに応じて自身の攻撃力を低下させる項目

;_______________________________________________________________________
;
;リコイル停止時間設定 Recoil Guard Pause Time
;_______________________________________________________________________

[state -3];リコイル停止時間設定
type = varset
trigger1 = 1
var(32) = 15  ;<--フレーム数で指定、最小2、最大22。デフォルトは20  minimum is 2 , maximum is 22.
ignorehitpause = 1


;要は画面が何フレーム止まるかです。性能自体は変化しません。
;原作では20Fぐらいです。ヒットポーズに合わせるなら10程度推奨。このぐらいなら比較的サクサク動きます。
;タッグ戦やリコイルLv10にする場合は5〜9程度でもいいかもしれません。最小の2だと早すぎる気がします。
;自分で使ってみていい感じの数値を入れてみてください。

;最大22とは書いていますが、ぶっちゃけ何フレームでもかまいません。
;ただ上限書いておかないと、とんでもない値入れる人がでてきたりして、それで「画面が動かない」とか言われても困るので念のためです。
;ちなみに下限はairの関係上の下限なのでそれより下の値は入れないでください。
